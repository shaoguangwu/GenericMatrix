762 template<typename Elem>
763 GenericMatrix<Elem> GenericMatrix<Elem>::roi(size_type row1, size_type col1, size_type row2, size_type col2)
764 {
765  if (!containsIndex(row1, col1) || !containsIndex(row2, col2)) {
766  return GenericMatrix<Elem>();
767  }
768  std::pair<size_type, size_type> rowMinMax = std::minmax(row1, row2);
769  std::pair<size_type, size_type> colMinMax = std::minmax(col1, col2);
770  GenericMatrix<Elem> result(rowMinMax.second - rowMinMax.first + 1, colMinMax.second - colMinMax.first + 1);
771  for (size_type i = rowMinMax.first; i <= rowMinMax.second; ++i) {
772  std::copy_n(constData(i, colMinMax.first), result.columns(), result.data((i - rowMinMax.first), 0));
773  }
774  return result;
775 }
